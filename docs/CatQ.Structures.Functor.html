Class Is_Functor
      (fdom fcod : Category)
      (fobj : fdom → fcod)
      (fmorphism : ∀ {a b}, @morphism fdom a b -⇒ @morphism fcod (fobj a) (fobj b)) :=
  {
    fmorphism_identity : ∀ {a}, fmorphism (@identity fdom a) == @identity fcod (fobj a);
    fmorphism_compose : ∀ {a b c} {f : a ⟶ b} {g : b ⟶ c}, fmorphism (compose (g,f)) == compose (fmorphism g, fmorphism f);
  }.

Structure Functor (fdom fcod : Category) :=

    is_functor :> Is_Functor (@fmorphism);
  }.

Definition fmap {C D : Category} (F : Functor C D) :
  ∀ {a b}, hom a b → hom (fobj F a) (fobj F b) := fun _ _ ⇒ fmorphism F.

Lemma fmap_identity {C D : Category} (F : Functor C D) :
  ∀ {a}, fmap F (@identity C a) == @identity D (F a).
Proof.
  unfold fmap. intro.
  apply fmorphism_identity.
Defined.

Lemma fmap_compose {C D : Category} (F : Functor C D) :
  ∀ {a b c} {g : b ⟶ c} {f : a ⟶ b}, fmap F (g ∘ f) == fmap F g ∘ fmap F f.
Proof.
  unfold fmap. intros.
  apply fmorphism_compose.
Defined.

Lemma fmap_proper {C D : Category} (F : Functor C D) {a b : C} :
  Proper (@equality _ ==> @equality _) (fmap F (a:=a) (b:=b)).
Proof.
  solve_proper.
Defined.

Structure Functor_Type (fdom fcod : Category) :=
  {
    funct_obj : fdom → fcod;
    funct_map : ∀ {a b}, hom a b → hom (funct_obj a) (funct_obj b);
    funct_map_proper : ∀ {a b}, Proper ((@equality _) ==> (@equality _)) (@funct_map a b);
    funct_identity : ∀ {a}, funct_map (@identity fdom a) == @identity fcod (funct_obj a);
    funct_compose : ∀ {a b c} {f : hom a b} {g : hom b c}, funct_map (g ∘ f) == funct_map g ∘ funct_map f;
  }.

  [fmap: fun _ _ f ⇒ f with fun a ⇒ a as C to C].

Program Definition compFunctor {C D E : Category} (G : Functor D E) (F : Functor C D) : Functor C E :=
  [fmap: fun _ _ f ⇒ fmap G (fmap F f) with fun a ⇒ fobj G (fobj F a)].
Next Obligation.
  solve_proper.
Defined.
Next Obligation.
  setoid_rewrite fmorphism_identity.
  setoid_rewrite fmorphism_identity.
  reflexivity.
Defined.
Next Obligation.
  setoid_rewrite fmorphism_compose.
  setoid_rewrite fmorphism_compose.
  reflexivity.
Defined.

Lemma compFunctor_compose {C D E} (G : Functor D E) (F : Functor C D) {a b} (f : a ⟶ b) :
  fmap (G ∘f F) f == fmap G (fmap F f).
Proof.
  reflexivity.
Qed.

Definition eqFunctor {C D} (F G : Functor C D) :=
  ∀ {a b} (f : a ⟶ b), fmap F f ≈ fmap G f.

Notation "F ==f G" := (eqFunctor F G) (at level 40).

Instance eqFunctor_equiv {C D} : Equivalence (@eqFunctor C D).
Proof.
  constructor.
  - unfold Reflexive.
    intro.
    constructor.
    reflexivity.
  - unfold Symmetric.
    intros.
    unfold eqFunctor.
    intros.
    unfold eqFunctor in H.
    destruct (H a b f).
    constructor.
    symmetry.
    exact H0.
  - unfold Transitive.
    intros.
    unfold eqFunctor.
    intros.
    destruct (H0 a b f).
    destruct (H a b f).
    constructor.
    rewrite H2, H1.
    reflexivity.
Defined.

Lemma eqFunctor_obj {C D} {F F' : Functor C D} : F ==f F' → ∀ a, F a = F' a.
Proof.
  intros.
  destruct (H a a identity).
  reflexivity.
Qed.

Definition full {C D : Category} (F : Functor C D) : Prop := ∀ {a b}, surj (@fmorphism C D F a b).
Definition faithful {C D : Category} (F : Functor C D) : Prop := ∀ {a b}, inj (@fmorphism C D F a b).

Definition ff {C D : Category} (F : Functor C D) : Type
  := ∀ {a b}, (morphism a b) ≃ (morphism (F a) (F b)) in Setoids.
